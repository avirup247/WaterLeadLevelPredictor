567 template <typename valueT>
568 void append_hexidecimal(const stream& os, const valueT& value) {
569  /* If value is negative, display the minus sign and convert the value to its
570  * absolute value. */
571  auto absValue = detail::abs_value<valueT>::get(os, value);
572 
573  char digits[MAX_HEX_DIGITS] = {0};
574  int currentDigit = 0;
575 
576  valueT recValue = absValue;
577  while (recValue > 0) {
578  digits[currentDigit] = recValue % 16;
579  recValue /= 16;
580  currentDigit++;
581  }
582 
583  os.append_chars(COMPUTECPP_HEX_PREFIX, 2);
584 
585  for (int i = (currentDigit - 1); i >= 0; i--) {
586  char c = (digits[i] <= 9) ? detail::to_char(digits[i])
587  : static_cast<char>('A' + (digits[i] - 10));
588  os.append_chars(&c, 1);
589  }
590 }

598 template <typename valueT>
599 inline int append_integral(const stream& os, const valueT& value,
600  bool useManipulators = true) {
601  if (value == 0) {
602  // A zero is always displayed
603  auto character = detail::to_char(0);
604  constexpr int numDigits = 1;
605  os.append_chars(&character, numDigits);
606  return numDigits;
607  }
608 
609  /* Handle hex manipulator. */
610  if (useManipulators && (os.get_stream_mode() == stream_mode::hex)) {
611  append_hexidecimal(os, value);
612  return 0;
613  }
614 
615  /* If value is negative, display the minus sign and convert the value to its
616  * absolute value. */
617  auto absValue = detail::abs_value<valueT>::get(os, value);
618 
619  /* Create a variable with the original value, that is divided by 10 until it
620  * reaches zero. */
621  valueT devidedValue = absValue;
622 
623  /* Create an array to store the digits retrieved from the value. */
624  valueT digits[MAX_INTEGRAL_DIGITS] = {0};
625 
626  /* Create a counter to record the number of digits retrieved form the value.
627  */
628  int numDigits = 0;
629 
630  /* While the divided value is still greater than zero */
631  while (devidedValue > 0) {
632  /* Calculate a multiplier based on the current digit counter. */
633  valueT multiplier = 10;
634  for (int m = 0; m < numDigits; m++) {
635  multiplier *= 10;
636  }
637 
638  /* Retrieve the digit based on the current digit counter, by dividing by the
639  * multiplier, multiplying by the multiplier, subtracting this from the
640  * original value and then finally dividing by the multiple divided by 10,
641  * reducing the value down to a single digit. */
642  valueT singleDigit =
643  (absValue - ((absValue / multiplier) * multiplier)) / (multiplier / 10);
644 
645  /* assign the digit retrieved. */
646  digits[numDigits] = singleDigit;
647 
648  /* Divided the value again and increment the digit counter. */
649  devidedValue /= 10;
650  numDigits++;
651  }
652 
653  /* Loop over the array of retrieved digits in reverse order, converting each
654  * to a character and appending it. */
655  for (int i = numDigits - 1; i >= 0; i--) {
656  auto integral = static_cast<int>(digits[i]);
657  auto character = detail::to_char(integral);
658  os.append_chars(&character, 1);
659  }
660 
661  /* Return the number fo digits. */
662  return numDigits;
663 }

671 template <typename valueT>
672 inline void append_floating_point(const stream& os, const valueT& value) {
673  /* If value is negative, display the minus sign and convert the value to its
674  * absolute value. */
675  auto absValue = detail::abs_value<valueT>::get(os, value);
676 
677  /* Create variables for storing the scientific base. */
678  int scientificBase = 0;
679 
680  /* Create a variable to storing whether the base is positive or not. */
681  bool possitiveBase = true;
682 
683  /* Handle scientific manipulator. */
684  if (os.get_stream_mode() == stream_mode::scientific) {
685  /* If value is greater than zero. */
686  if (absValue >= 0) {
687  /* Divide the value by 10 until it reaches a single digit on the integral
688  * side. */
689  while (absValue > valueT(10)) {
690  absValue /= valueT(10);
691  scientificBase++;
692  }
693  }
694  /* If value is less than zero. */
695  else {
696  /* Multiply the value by 10 until it reaches a single digit on the
697  * integral side. */
698  while (absValue < valueT(0.1)) {
699  absValue *= valueT(10);
700  scientificBase--;
701  }
702  possitiveBase = false;
703  }
704  }
705 
706  /* Cast to an integer to retrieve the integral part of the floating point. */
707  int integralPart = static_cast<int>(absValue);
708 
709  /* Append the integral part pf the floating point. */
710  int digitsInIntegralPart = append_integral<int>(os, integralPart);
711 
712  /* Append the period sign. */
713  os.append_chars(COMPUTECPP_DECIMAL_POINT, 1);
714 
715  /* Subtract the integral part from the original value to retrieve the
716  * fractional part. */
717  valueT fractionalPart = absValue - static_cast<valueT>(integralPart);
718 
719  /* Create an array to store the digits retrieved from the value. */
720  valueT digits[MAX_FRACTIONAL_DIGITS] = {valueT{0}};
721 
722  /* Retrieve the precision. */
723  int precision = os.get_precision();
724 
725  /* Calculate digits for fractional side. */
726  int digitsForFractionalSide = precision - digitsInIntegralPart;
727 
728  /* Loop for each digit of precision. */
729  for (int i = 0; i < (digitsForFractionalSide - 1); i++) {
730  /* Create a multiplier based on the current digit. */
731  auto currentMultiplier = valueT{10.0};
732  for (int m = 0; m < i; m++) {
733  currentMultiplier *= valueT{10.0};
734  }
735 
736  /* Multiply the fractional part by the multiplier in order to convert the
737  * current digit in the fractional part to a digit on the integral part. */
738  valueT multipliedFloat = fractionalPart * currentMultiplier;
739 
740  /* Loop for each previously retrieved digit, creating a multiplier for it
741  * and
742  * using that to remove all integral parts beyond the first digit. */
743  for (int r = 0; r < i; r++) {
744  /* Create a multiplier based on the difference in digits between the
745  * current loop and the current digit. */
746  float currentStageMultiplier = 1.0f;
747  for (int sm = 0; sm < (i - r); sm++) {
748  currentStageMultiplier *= 10.0f;
749  }
750 
751  /* Multiply the multiplier by the stored digit to obtain the value to
752  * remove from the integral part. */
753  multipliedFloat -=
754  (digits[r] * static_cast<valueT>(currentStageMultiplier));
755  }
756 
757  /* Assign the retrieved digit to the array */
758  digits[i] = static_cast<valueT>((static_cast<int>(multipliedFloat)));
759 
760  /* Convert the digit to an character and append it. */
761  auto integral = static_cast<int>(digits[i]);
762  auto character = detail::to_char(integral);
763  os.append_chars(&character, 1);
764  }
765 
766  /* If scientific manipulator is enabled add the scientific base. */
767  if (os.get_stream_mode() == stream_mode::scientific) {
768  /* Append the scientific notation. */
769  if (possitiveBase) {
770  os.append_chars(COMPUTECPP_POSITIVE_EXP_PREFIX, 2);
771  } else {
772  os.append_chars(COMPUTECPP_NEGATIVE_EXP_PREFIX, 2);
773  }
774 
775  /* Create variable to store the number of digits in the scientific base. */
776  int digitsInScientificBase = 0;
777 
778  /* Divide the scientific base by 10 until it reaches zero to detect the
779  * number of digits. */
780  int tempScientificBase = scientificBase;
781  while (tempScientificBase > 0) {
782  tempScientificBase /= 10;
783  digitsInScientificBase++;
784  }
785 
786  /* Add the correct number of zeros before the scientific base. */
787  for (int sbd = 0; sbd < (3 - digitsInScientificBase); sbd++) {
788  os.append_chars("0", 1);
789  }
790 
791  /* Appen the integral specifying the scientific base. */
792  append_integral(os, scientificBase);
793  }
794 }
