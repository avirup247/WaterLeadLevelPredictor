218  template <typename workItemFunctionT>
219  void parallel_for_work_item(range<dimensions> flexibleRange_,
220  const workItemFunctionT& func) const {
221  const auto globalRange = this->get_global_range();
222  const auto physicalLocalRange =
223  detail::index_array{this->get_local_range()};
224  const auto groupID = detail::index_array{this->get_id()};
225  const auto globalIdBase = physicalLocalRange * groupID;
226 
227  detail::index_array physicalLocalID;
228  const auto flexibleRange = detail::index_array{flexibleRange_};
229 
230  for (size_t itemZ = 0; itemZ < flexibleRange[2]; itemZ++) {
231  physicalLocalID[2] = itemZ % physicalLocalRange[2];
232  for (size_t itemY = 0; itemY < flexibleRange[1]; itemY++) {
233  physicalLocalID[1] = itemY % physicalLocalRange[1];
234  for (size_t itemX = 0; itemX < flexibleRange[0]; itemX++) {
235  physicalLocalID[0] = itemX % physicalLocalRange[0];
236 
237  id<dimensions> localID(detail::index_array(itemX, itemY, itemZ));
238  id<dimensions> globalID = globalIdBase + physicalLocalID;
239 
240  h_item<dimensions> currentItem(detail::h_item_base(
241  detail::item_base(localID, flexibleRange),
242  detail::item_base(physicalLocalID, physicalLocalRange),
243  detail::item_base(globalID, globalRange)));
244 
245  func(currentItem);
246  }
247  }
248  }
