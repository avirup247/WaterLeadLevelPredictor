303 /* clone_data.
304  * @brief Copies a range of iterators into a temporary memory address
305  * allocated using an user-defined allocator.
306  * A deleter is set to call the user-define deallocate function.
307  * @tparam T Type of the underlying data
308  * @tparam AllocatorT Type of user supplied allocator
309  * @param begin Start of the range
310  * @param end End of the range
311  * @param alloc Allocator object to be used
312  * @return type-erased shared pointer
313  */
314 template <typename T, typename AllocatorT, typename Iterator>
315 shared_ptr_class<void> clone_data(Iterator begin, Iterator end,
316  AllocatorT alloc = AllocatorT()) {
317  size_t size = std::distance(begin, end);
318  auto typedAlloc = wrapped_allocator<AllocatorT>(
319  alloc, make_base_allocator<T, AllocatorT>::get_elem_size_multiplier());
320  shared_ptr_class<void> p(typedAlloc.allocate(size),
321  [typedAlloc, size](void* f) mutable {
322  typedAlloc.deallocate(static_cast<T*>(f), size);
323  });
324  std::copy(begin, end, static_cast<T*>(p.get()));
325  return p;
326 }
