593  accessor<elemT, 1, kMode, kTarget, access::placeholder::true_t>
594  get_accessor_with_added_offset(int addedOffset) const {
595  detail::access_range range = this->get_access_range();
596  // ensure that negative values move the offset backwards
597  intptr_t result = static_cast<intptr_t>(addedOffset) +
598  static_cast<intptr_t>(range.offset[0]);
599  // check underflow
600  if (result < 0) {
601  COMPUTECPP_CL_ERROR_CODE_MSG(
602  CL_SUCCESS, detail::cpp_error_code::ACCESSOR_ARGUMENTS_ERROR, nullptr,
603  "Attempted arithmetic operation out of accessor bounds. Underflow");
604  }
605  // check overflow
606  size_t resultSizeT = static_cast<size_t>(result);
607  if (resultSizeT + range.range[0] > this->get_store_range()[0]) {
608  COMPUTECPP_CL_ERROR_CODE_MSG(
609  CL_SUCCESS, detail::cpp_error_code::ACCESSOR_ARGUMENTS_ERROR, nullptr,
610  "Attempted arithmetic operation out of accessor bounds. Overflow");
611  }
612  // set the result
613  range.offset[0] = resultSizeT;
614 
615  cl::sycl::storage_mem storage(this->get_store());
616  return accessor<elemT, 1, kMode, kTarget, access::placeholder::true_t>(
617  storage, range);
618  }
