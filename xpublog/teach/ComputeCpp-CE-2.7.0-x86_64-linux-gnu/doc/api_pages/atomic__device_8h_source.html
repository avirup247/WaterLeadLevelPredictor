318 // OpenCL 1.2 does not implement compare_and_exchange in the same way as C++11,
319 // so this function emulates the behavior on the device. It attempts to figure
320 // out if the call to cmpxchg actually altered the atomic value by capturing the
321 // old value then comparing it to expected. If they are the same, then the
322 // comparison will have succeeded, so we can return true. If they aren't, then
323 // comparison failed, so we need to update expected manually then return false.
324 // The helper function performs this part of the logic, since it is shared by
325 // all compare_exchange device implementations.
326 template <typename T, access::address_space AddressSpace>
327 inline cl_bool atomic<T, AddressSpace>::compare_exchange_strong(
328  T& expected, T desired, memory_order /*success*/,
329  memory_order /*fail*/) const noexcept {
330  auto old = detail::atomic_helper<T>::cmpxchg(m_data, expected, desired);
331  return ::cl::sycl::detail::cmpxchg_helper(old, expected);
332 }
