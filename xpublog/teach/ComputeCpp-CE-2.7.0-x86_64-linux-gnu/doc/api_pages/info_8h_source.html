501 template <typename syclInfo, typename syclType, typename clType,
502  cl_bitfield andBits>
503 syclType get_opencl_info_as_sycl(
504  const typename opencl_info_base<syclInfo>::cl_object& clObject,
505  cl_uint clParam) {
506  /* Create an instance of the SYCL type to be returned. */
507  auto syclValue = syclType();
508 
509  /* Create a variable for the size of the value that will be returned. */
510  size_t size = 0;
511 
512  /* Retrieve the size of the OpenCL info value that will be returned. */
513  get_opencl_info<syclInfo>(clObject, clParam, nullptr, 0, &size);
514 
515  /* If zero size, there is nothing else to be done. */
516  if (size == 0) {
517  return syclValue;
518  }
519 
520  /* Allocate memory for the char* based on the size of the value that will be
521  * returned. */
522  vector_class<char> charVector(size);
523 
524  /* Retrieve the value for the OpenCL info parameter specified by clParam. */
525  get_opencl_info<syclInfo>(clObject, clParam, charVector.data(), size,
526  nullptr);
527 
528  /* Calculate the number of elements that were returned. */
529  size_t numElems = size / sizeof(clType);
530 
531  /* Cast the char* to a pointer to the corresponding OpenCL type. */
532  clType* clPtr = reinterpret_cast<clType*>(charVector.data());
533 
534  /* Do we need to extract any bits from the clValue before casting to the SYCL
535  type? */
536  extract_bit<clType, andBits>::extract(clPtr);
537 
538  /* Convert the pointer to the OpenCL type to the corresponding SYCL type. */
539  syclValue =
540  info_convert<clType*, syclType>::cl_to_sycl(clPtr, numElems, clParam);
541 
542  /* Return the SYCL value. */
543  return syclValue;
544 }

592 template <typename syclInfo, typename syclType, typename clType,
593  cl_uint clParam, cl_bitfield andBits>
594 syclType get_sycl_info(
595  const typename opencl_info_base<syclInfo>::cl_object& clObject,
596  bool isHost) {
597  /* If the requesting object is in host mode retrieve the info through the host
598  * info definitions. */
599  if (use_host_info_definitions(isHost)) {
600  /* Retrieve the SYCL info host info specified by clParam. */
601  return sycl_host_info<syclType, clParam>::get();
602  }
603  /* If the requesting object is not in host mode retrieve the info through the
604  OpenCL info API and convert it to the corresponding SYCL type. */
605  else {
606  return detail::get_opencl_info_as_sycl<syclInfo, syclType, clType, andBits>(
607  clObject, clParam);
608  }
609 }

626 inline string_class make_valid_string(const char* clPtr,
627  const size_t numElems) {
628  const auto validString = clPtr && numElems != 0;
629 
630  // return empty string object if the argument is empty
631  if (!validString) {
632  return {};
633  }
634 
635  // We can't be sure the clPtr argument is null terminated, so the object
636  // is created, and if a terminating byte is detected, it is removed.
637 
638  string_class result(clPtr, numElems);
639  if (!result.empty() && result.back() == '\0') {
640  result.pop_back();
641  }
642  return result;
643 }

646 inline std::vector<std::string>
647 info_convert<char*, std::vector<std::string>>::cl_to_sycl(const char* clPtr,
648  size_t /*numElems*/,
649  cl_uint /*clParam*/) {
650  auto inputString = std::string{clPtr};
651 
652  // Replace semi-colons with spaces as some implementations use semi-colons
653  // and some use spaces.
654  std::replace(std::begin(inputString), std::end(inputString), ';', ' ');
655 
656  // Copy each space delimited substring from the input string into the
657  // result vector.
658  std::istringstream inputStream{inputString};
659 
660  std::vector<std::string> result(
661  std::istream_iterator<std::string>{inputStream},
662  std::istream_iterator<std::string>{});
663 
664  if (!inputStream.eof()) {
665  COMPUTECPP_CL_ERROR_CODE_MSG(
666  CL_SUCCESS, detail::cpp_error_code::GET_INFO_ERROR, nullptr,
667  "Error separating extensions into individual strings.");
668  }
669 
670  return result;
671 }
