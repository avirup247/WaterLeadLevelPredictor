364 COMPUTECPP_CONSTEXPR_CPP14 CRTP id_range_binary_op(CRTP lhs, const CRTP& rhs,
365  const Func& func) noexcept {
366  using dataT = typename CRTP::value_type;
367  constexpr const auto dims = CRTP::dimensions;
368  for (int i = 0; i < dims; ++i) {
369  lhs[i] = static_cast<dataT>(func(lhs[i], rhs[i]));
370  }
371  return lhs;
372 }

374 template <class Func, class CRTP>
375 COMPUTECPP_CONSTEXPR_CPP14 bool id_range_comparison_op(
376  const CRTP& lhs, const CRTP& rhs, const Func& func) noexcept {
377  constexpr const auto dims = CRTP::dimensions;
378  bool ret = func(lhs[0], rhs[0]);
379  for (int i = 1; i < dims; ++i) {
380  ret = ret && func(lhs[i], rhs[i]);
381  }
382  return ret;
383 }
