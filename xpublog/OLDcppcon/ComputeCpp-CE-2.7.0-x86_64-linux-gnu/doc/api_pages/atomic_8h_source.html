435 template <typename T, access::address_space addressSpace>
436 inline T atomic<T, addressSpace>::fetch_min(const T operand,
437  const memory_order memOrder) const
438  noexcept {
439  /* Standard C++ 11 defines no "min" operation, so this function emulates the
440  * behavior by executing a loop. First, the value is loaded, then compared
441  * against the operand. After this a compare_exchange is used to check that
442  * the value hasn't been updated sneakily (by another thread say); if it has
443  * (i.e. m_data doesn't equal its old value) store the new value of the atomic
444  * and try again. */
445  T old = m_data->load(std::memory_order_relaxed);
446  do {
447  if (old < operand) {
448  break;
449  }
450  } while (!compare_exchange_weak(old, operand, memOrder, memOrder));
451  return old;
452 }
