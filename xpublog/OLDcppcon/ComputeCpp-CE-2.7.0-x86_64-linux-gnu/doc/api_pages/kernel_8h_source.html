335 template <typename InfoEnum, InfoEnum EnumValue, typename... Args>
336 typename info::param_traits<InfoEnum, EnumValue>::
337  return_type inline get_kernel_info_impl(const dkernel_shptr& context,
338  const cl_device_id device,
339  const Args&... args) {
340  using cl_type = typename opencl_kernel_ext_info<InfoEnum, EnumValue>::cl_type;
341  using sycl_type =
342  typename opencl_kernel_ext_info<InfoEnum, EnumValue>::sycl_type;
343 
344  constexpr auto clParam =
345  opencl_kernel_ext_info<InfoEnum, EnumValue>::cl_param;
346 
347  constexpr auto numOutputElems =
348  opencl_kernel_ext_info<InfoEnum, EnumValue>::cl_type_num_elems;
349  constexpr auto outputSize = sizeof(cl_type) * numOutputElems;
350 
351  auto oclInputPair = convert_sycl_to_ocl(args...);
352  auto inputPtr = oclInputPair.first;
353  auto inputSize = oclInputPair.second;
354 
355  char buffer[outputSize];
356  get_opencl_kernel_info<InfoEnum>(context, device, clParam, inputSize,
357  inputPtr, outputSize, buffer);
358  cl_type* clPtr = reinterpret_cast<cl_type*>(buffer);
359  return info_convert<cl_type*, sycl_type>::cl_to_sycl(clPtr, numOutputElems,
360  clParam);
361 }
