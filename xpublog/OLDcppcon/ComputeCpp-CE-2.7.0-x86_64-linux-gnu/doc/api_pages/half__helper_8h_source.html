53 COMPUTECPP_CONSTEXPR_CPP14 Float32 toFloat(Half16 h) {
54  Float32 o = {0};
55 
56  // From ISPC ref code
57  if (h.Exponent == 0u && h.Mantissa == 0u) { // (Signed) zero
58  o.Sign = h.Sign;
59  } else {
60  if (h.Exponent == 0u) // Denormal (will convert to normalized)
61  {
62  // Adjust mantissa so it's normalized (and keep track of exp adjust)
63  int e = -1;
64  unsigned m = h.Mantissa;
65  do {
66  e++;
67  m <<= 1;
68  } while ((m & 0x400) == 0);
69 
70  o.Mantissa = (m & 0x3ff) << 13;
71  o.Exponent = 127 - 15 - e;
72  o.Sign = h.Sign;
73  } else if (h.Exponent == 0x1fu) // Inf/NaN
74  {
75  // NOTE: It's safe to treat both with the same code path by just
76  // truncating lower Mantissa bits in NaNs (this is valid).
77  o.Mantissa = static_cast<unsigned int>(h.Mantissa) << 13;
78  o.Exponent = 255;
79  o.Sign = h.Sign;
80  } else // Normalized number
81  {
82  o.Mantissa = static_cast<unsigned int>(h.Mantissa) << 13u;
83  o.Exponent = 127u - 15u + h.Exponent;
84  o.Sign = h.Sign;
85  }
86  }
87 
88  return o;
89 }

95 COMPUTECPP_CONSTEXPR_CPP14 Half16 toHalf(Float32 f) {
96  Half16 o = {0};
97 
98  // Based on ISPC reference code (with minor modifications)
99  if (f.Exponent == 0u) { // Signed zero/denormal (which will underflow)
100  o.Exponent = 0;
101  } else if (f.Exponent == 255u) // Inf or NaN (all exponent bits set)
102  {
103  o.Exponent = 31;
104  o.Mantissa = f.Mantissa ? 0x200 : 0; // NaN->qNaN and Inf->Inf
105  } else // Normalized number
106  {
107  // Exponent unbias the single, then bias the halfp
108  int newexp = static_cast<int>(f.Exponent) - 127 + 15;
109  if (newexp >= 31) // Overflow, return signed infinity
110  o.Exponent = 31;
111  else if (newexp <= 0) // Underflow
112  {
113  if ((14 - newexp) <= 24) // Mantissa might be non-zero
114  {
115  unsigned mant = f.Mantissa | 0x800000u; // Hidden 1 bit
116  o.Mantissa = mant >> (14 - newexp);
117  if ((mant >> (13 - newexp)) & 1) // Check for rounding
118  o.u++; // Round, might overflow into exp bit, but this is OK
119  }
120  } else {
121  o.Exponent = newexp;
122  o.Mantissa = static_cast<unsigned int>(f.Mantissa) >> 13u;
123  if (f.Mantissa & 0x1000u) // Check for rounding
124  o.u++; // Round, might overflow to inf, this is OK
125  }
126  }
127 
128  o.Sign = f.Sign;
129  return o;
130 }
