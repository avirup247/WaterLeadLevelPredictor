486  static program create_program_for_kernel(cl::sycl::context c) {
487  if (detail::kernel_info<kernelT>::name == nullptr && !c.is_host()) {
488  COMPUTECPP_CL_ERROR_CODE_MSG(

625  template <typename BIType>
626  static const BIType& select_kernel_binary_info_helper(const BIType* binList,
627  size_t binListSize,
628  ddevice_wkptr dev) {
629  if ((binList == nullptr) || (binListSize == 0)) {
630  COMPUTECPP_CL_ERROR_CODE_MSG(
631  0, detail::cpp_error_code::BINARY_NOT_FOUND_ERROR, nullptr,
632  "Unable to retrieve a binary, is integration header included?");
633  }
634 
635  // Find most suitable binary
636  const auto endOfList = binList + binListSize;
637  auto bestMatch = endOfList;
638  int bestMatchValue = -1;
639  for (auto it = binList; it < endOfList; ++it) {
640  if (it->binary_info == nullptr) {
641  // This can happen when -fsycl-split-modules is used
642  break;
643  }
644  const int currentMatchValue =
645  rank_binary_info_impl(*it->binary_info, dev);
646  if (currentMatchValue > bestMatchValue) {
647  bestMatchValue = currentMatchValue;
648  bestMatch = it;
649  }
650  }
651 
652  if (bestMatchValue >= 0) {
653  // Only select positive matches
654  return *bestMatch;
655  }
656 
657  // The default case is to return the first kernel binary available.
658  return binList[0];
659  }
