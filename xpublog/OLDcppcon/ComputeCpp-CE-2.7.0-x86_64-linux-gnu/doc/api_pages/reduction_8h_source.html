780 template <int dimensions, typename userFuncT, typename reductionFuncT>
781 auto get_reduction_kernel_atomics(handler& cgh, userFuncT userFunc,
782  reductionFuncT reductionFunc,
783  const range<dimensions>& range,
784  const nd_range<dimensions>& ndRange) {
785  size_t wgSize = ndRange.get_local_range().size();
786  size_t numWorkgroups = ndRange.get_group_range().size();
787 
788  const bool initializeToIdentity = reductionFunc.initialize_to_identity();
789  auto groupSum = reductionFunc.get_rw_local_acc(1, cgh);
790  auto out = reductionFunc.get_user_storage();
791 
792  auto partialSums = reductionFunc.get_partial_sums_acc(numWorkgroups, cgh);
793  auto numWorkgroupsFinished = reductionFunc.get_workgroup_finished_acc(cgh);
794  auto doFinalWriteInLastWG =
795  reductionFunc.template get_rw_local_acc<int>(1, cgh);
796 
797  auto binaryOP = typename reductionFuncT::binary_operation{};
798 
799  return [userFunc, range, groupSum, out, initializeToIdentity, partialSums,
800  numWorkgroupsFinished, doFinalWriteInLastWG, binaryOP, numWorkgroups,
801  wgSize](nd_item<1> id) {
802  typename reductionFuncT::reducer_wrapper_t redu;
803 
804  reduction_loop<typename reductionFuncT::reducer_t>(range, redu, id,
805  userFunc);
806  size_t linearID = id.get_local_linear_id();
807  if (linearID == 0) {
808  groupSum[0] = redu.identity();
809  }
810  group_barrier(id.get_group(), memory_scope::work_group);
811  redu.template atomic_combine<access::address_space::local_space>(
812  &groupSum[0]);
813  group_barrier(id.get_group(), memory_scope::work_group);
814  if (linearID == 0) {
815  partialSums[id.get_group_linear_id()] = groupSum[0];
816 
817  auto numFinished = atomic<int, access::address_space::global_space>(
818  numWorkgroupsFinished.get_pointer());
819  numFinished.fetch_add(1);
820  doFinalWriteInLastWG[0] =
821  numFinished.load() == static_cast<int>(numWorkgroups);
822  groupSum[0] = redu.identity();
823  }
824  group_barrier(id.get_group(), memory_scope::work_group);
825  if (doFinalWriteInLastWG[0]) {
826  for (size_t i = linearID; i < numWorkgroups; i += wgSize) {
827  redu.set_value(partialSums[i]);
828  redu.template atomic_combine<access::address_space::local_space>(
829  &groupSum[0]);
830  }
831  group_barrier(id.get_group(), memory_scope::work_group);
832  if (linearID == 0) {
833  if (!initializeToIdentity) {
834  groupSum[0] =
835  binaryOP(groupSum[0], reductionFuncT::get_out_pointer(out)[0]);
836  }
837  reductionFuncT::get_out_pointer(out)[0] = groupSum[0];
838  }
839  }
840  };
841 }

889  /* Performs a tree-reduction inside each workgroup to reduce down to a
890  * partial sum for each group. localSums[wgSize] reduces odd elements
891  * when the step size is odd.
892  */
893  size_t stepSize = wgSize;
894  for (size_t currentStep = stepSize / 2; currentStep > 0; currentStep /= 2) {
895  if (linearID < currentStep) {
896  localSums[linearID] =
897  binaryOP(localSums[linearID], localSums[linearID + currentStep]);
898  } else if (linearID == currentStep && (stepSize & 0x1)) {
899  localSums[wgSize] =
900  binaryOP(localSums[wgSize], localSums[stepSize - 1]);
901  }
902  group_barrier(id.get_group(), memory_scope::work_group);
903  stepSize = currentStep;
904  }
905 
906  /* Writes the partial sum to global memory. Uses atomic to sync between
907  * workgroups.
908  */
909  if (linearID == 0) {
910  auto value = binaryOP(localSums[0], localSums[wgSize]);
911 
912  partialSums[id.get_group_linear_id()] = value;
913 
914  auto numFinished = atomic<int, access::address_space::global_space>(
915  numWorkgroupsFinished.get_pointer());
916  numFinished.fetch_add(1);
917  doPartialSumInLastWG[0] =
918  numFinished.load() == static_cast<int>(numWorkgroups);
919  }
920  group_barrier(id.get_group(), memory_scope::work_group);
921 
922  /* Do a final reduction in the last workgroup, summing the
923  * partial sums from other workgroups and writing out to the user memory.
924  */
925  if (doPartialSumInLastWG[0]) {
926  auto localSum = redu.identity();
927  for (size_t i = linearID; i < numWorkgroups; i += wgSize) {
928  localSum = binaryOP(localSum, partialSums[i]);
929  }
930  localSums[linearID] = localSum;
931  if (linearID == 0) {
932  localSums[wgSize] = redu.identity();
933  }
934  group_barrier(id.get_group(), memory_scope::work_group);
935 
936  stepSize = wgSize;
937  for (size_t currentStep = stepSize / 2; currentStep > 0;
938  currentStep /= 2) {
939  if (linearID < currentStep) {
940  localSums[linearID] =
941  binaryOP(localSums[linearID], localSums[linearID + currentStep]);
942  } else if (linearID == currentStep && (stepSize & 0x1)) {
943  localSums[wgSize] =
944  binaryOP(localSums[wgSize], localSums[stepSize - 1]);
945  }
946  group_barrier(id.get_group(), memory_scope::work_group);
947  stepSize = currentStep;
948  }
949  if (linearID == 0) {
950  auto value = binaryOP(localSums[0], localSums[wgSize]);
951  if (!initializeToIdentity) {
952  value = binaryOP(value, reductionFuncT::get_out_pointer(out)[0]);
953  }
954  reductionFuncT::get_out_pointer(out)[0] = value;
955  }
956  }
957  };
958 }
